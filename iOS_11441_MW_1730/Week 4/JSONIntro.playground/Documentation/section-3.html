class Contact: Printable {

    var firstName: String
    var lastName: String
    var email: String
    var cell: String

    init(contactDictionary: NSDictionary) {
        firstName = (contactDictionary["firstName"] ?? "") as String
        lastName = (contactDictionary["lastName"] ?? "") as String
        email = (contactDictionary["email"] ?? "") as String
        cell = (contactDictionary["cell"] ?? "") as String
    }

    var description: String {
        return "\(firstName) \(lastName), \(email), \(cell)"
    }
}

We access each element in the dictionary with code like this.

1
contactDictionary["firstName"] ?? ""

We have to do this because Swift doesn't know if there is a value for any given key. So it always returns an Optional. We use the ?? to check if there is a value. If not we'll get "", and empty string. Safety first!

Because the contactDictionary is an NSDictionary we now need to cast it to a Swift string with as String. That gives us the full syntax.

1
(contactDictionary["firstName"] ?? "") as String
