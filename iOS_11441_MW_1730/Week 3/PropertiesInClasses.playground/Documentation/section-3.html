class Point {
    var pointX: Double
    var pointY: Double
    var point: (Double, Double) {
        get {
            return (pointX, pointY)
        }
        set {
            pointX = newValue.0
            pointY = newValue.1
        }
    }

    init() {
        pointX = 0.0
        pointY = 0.0
    }

    init(x: Double, y: Double) {
        pointX = x
        pointY = y
    }
}

2

println("Point: x:\(point1.pointX), y:\(point1.pointY)")

Read-Only Computed Properties

If we want a read-only computer property we only make the get. Remember the description() functions we've been writing? Let's do that with a property. For this we just make the get portion of the code.

1
2
3
4
5
    var description: String {
        get {
            return "Point: x:\(pointX), y:\(pointY)"
        }
    }

Which lets us do this.

1
2
println(point1.description)
println(point2.description)

Which gives us this.

1
2
Point: x:10.0, y:20.0
Point: x:100.0, y:150.0
